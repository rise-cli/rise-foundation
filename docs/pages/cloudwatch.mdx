`

    name: 'myDashboard',
    rows: [
        {
            type: 'LAMBDAROW',
            verticalPosition: 0,
            name: 'nameOfRow',
            region: 'us-east-1',
            functionName: `functionName`,
            docs: markdown,

            invocationGoal: 10,
